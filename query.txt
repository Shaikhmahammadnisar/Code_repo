how to identify duplicate records in mysql 

 select *,count(*) from emp group by id,name,gender,city,sal
    having count(*) > 1;


how to remove duplicate 

WITH ranked AS (
  SELECT id, 
         ROW_NUMBER() OVER (PARTITION BY id ORDER BY id) AS rn
  FROM emp
)
DELETE FROM emp
WHERE id IN (
    SELECT id FROM ranked WHERE rn > 1)


solution using row_number 

select * from
   (
   select id,name,gender,city,sal,row_number() over(partition by id,name,gender,city,sal order by  id) as rk from emp)
     as x where x.rk=1;
